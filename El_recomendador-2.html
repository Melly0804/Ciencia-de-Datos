    n = song_ratings.count()

def create_X(df):
    N = df["userId"].nunique()
    M = df["song_id"].nunique()

    user_mapper = dict(zip(np.unique(df["userId"]), list(range(N))))
    song_mapper = dict(zip(np.unique(df["song_id"]), list(range(M))))

    user_inv_mapper = dict(zip(list(range(N)), np.unique(df["userId"])))
    song_inv_mapper = dict(zip(list(range(M)), np.unique(df["song_id"])))

    user_index = [user_mapper[i] for i in df["userId"]]
    song_index = [song_mapper[i] for i in df["song_id"]]

    X = csr_matrix((df["rating"], (song_index, user_index)), shape=(M, N))

    return X, user_mapper, song_mapper, user_inv_mapper, song_inv_mapper

def find_similar_songs(song_id, X, k=5, metric="cosine", show_distance=False):
    song_ind = song_mapper[song_id]
    song_vec = X[song_ind]

    n_neighbors = k + 1

    kNN = NearestNeighbors(n_neighbors=n_neighbors, algorithm="brute", metric=metric)
    kNN.fit(X)

    if isinstance(song_vec, np.ndarray):
        song_vec = song_vec.reshape(1, -1)

    distances, indices = kNN.kneighbors(song_vec, return_distance=True)

    distances = distances.flatten()[1:]
    indices = indices.flatten()[1:]

    similar_song_ids = [song_inv_mapper[i] for i in indices]

    if show_distance:
        return list(zip(similar_song_ids, distances))
    else:
        return similar_song_ids

def recommend_for_user(user_id, k_neighbors=5, min_rating=4, top_n=10):
    user_ratings = ratings[ratings["userId"] == user_id].copy()

    if user_ratings.empty:
        raise ValueError("Ese user_id no existe en los datos.")

    already_rated = set(user_ratings["song_id"])
    liked = user_ratings[user_ratings["rating"] >= min_rating]

    if liked.empty:
        return pd.DataFrame(columns=["song_id", "song_title", "song_genre", "score"])

    scores = {}

    for _, row in liked.iterrows():
        source_song = row["song_id"]
        source_rating = row["rating"]

        neighbors = find_similar_songs(source_song, X, k=k_neighbors, show_distance=True)

        for neighbor_id, distance in neighbors:
            if neighbor_id in already_rated:
                continue

            similarity = 1 - distance
            scores[neighbor_id] = scores.get(neighbor_id, 0) + (source_rating * similarity)

    if not scores:
        return pd.DataFrame(columns=["song_id", "song_title", "song_genre", "score"])

    recs = pd.DataFrame([{"song_id": sid, "score": score} for sid, score in scores.items()])
    recs = recs.merge(catalogo, on="song_id", how="left")
    recs = recs.sort_values("score", ascending=False).head(top_n)

    return recs[["song_id", "song_title", "song_genre", "score"]]
